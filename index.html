Variables

Ejemplo 1 
/*Si declaro una variable con la palabra reservada var, aunque en otro scope 
le cambie el valor, el contenido de la variable se ha cambiado en todo el programa*/

//x es 10
var x = 10;
let y = 10;
document.write("El valor de var x es: "+ x);
document.write("El valor de let y es: "+ y);
{  
  x = 2;
  let y = 2;//Variable local del scope
  document.write("El valor de x dentro del ámbito es: "+ x);
  document.write("El valor de y dentro del ámbito es: "+ y);
  
  x = 5;//Si modifico el valor de una variable global, lo modifico en todo el programa:x = 5
  y = 5;
  document.write("El valor de x dentro del ámbito es: "+ x);
  document.write("El valor de y dentro del ámbito es: "+ y);
}
//x = 5
document.write("El valor de x fuera del ámbito es: "+ x);
//y = 10
document.write("El valor de y fuera del ámbito es: "+ y);

Bucle For

var cars = ["Saab", "Volvo", "BMW", "Mercedes"];
console.log(cars);
console.log(cars[0]);
console.log(cars[1]);
console.log(cars[2]);
console.log(cars[3]);

console.log("Contenido del vector recorrido mediante un for:");

let size = cars.length;
for(let i = 0; i < size; i++){
  console.log(cars[i]);//i contiene cada posición del array, uno para cada ciclo del for
}

/*Para recorrer un array es más cómodo usar un for of*/
console.log("Contenido del vector recorrido mediante un for of:");
for(let car of cars){
    console.log(car);
}

Bucle While

let aleatorio = Math.ceil((Math.random() * 10));
let escogido;

do{
  escogido = prompt("Escoge un número: ");
}while(aleatorio != escogido);

alert("Enhorabuena el número era el " + aleatorio);
